/**
 * t() — Vertaalfunctie
 * Gebruik: t('hero.client.headline') → string in huidige taal
 */
function t(key, lang = getCurrentLang()) {
  const parts = key.split('.');
  let node = translations;
  for (const part of parts) {
    node = node?.[part];
    if (!node) return `[${key}]`; // Fallback
  }
  return node?.[lang] ?? node?.['en'] ?? key;
}

function getCurrentLang() {
  return localStorage.getItem('buildpay_lang') || 'nl';
}

function setLang(lang) {
  localStorage.setItem('buildpay_lang', lang);
  document.documentElement.lang = lang;
  updateAllTranslations();
}

// Auto-update alle [data-i18n] elementen in de DOM
function updateAllTranslations() {
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.dataset.i18n;
    el.textContent = t(key);
  });
  // Placeholders
  document.querySelectorAll('[data-i18n-placeholder]').forEach(el => {
    el.placeholder = t(el.dataset.i18nPlaceholder);
  });
}

module.exports = { translations, t, setLang, getCurrentLang };

// React hook voor i18n — gebruik in alle componenten

import { useState, useEffect, useCallback } from 'react';
import { translations } from '../i18n/i18n';
import { getCountryConfig } from '../i18n/countries';

/**
 * useTranslation hook
 * 
 * Gebruik in componenten:
 * const { t, lang, country, setLang, setCountry, config } = useTranslation();
 * 
 * <h1>{t('hero.client.headline')}</h1>
 * <p>{t('common.getStarted')}</p>
 */
export function useTranslation() {
  const [lang, setLangState] = useState(
    () => localStorage.getItem('bp_lang') || detectBrowserLang()
  );
  const [country, setCountryState] = useState(
    () => localStorage.getItem('bp_country') || 'BE'
  );

  const config = getCountryConfig(country);

  // Vertaalfunctie met dot-notatie
  const t = useCallback((key) => {
    const parts = key.split('.');
    let node = translations;
    for (const part of parts) {
      node = node?.[part];
      if (!node) return `[${key}]`;
    }
    return node?.[lang] ?? node?.en ?? key;
  }, [lang]);

  const setLang = useCallback((newLang) => {
    localStorage.setItem('bp_lang', newLang);
    document.documentElement.lang = newLang;
    setLangState(newLang);
  }, []);

  const setCountry = useCallback((newCountry) => {
    localStorage.setItem('bp_country', newCountry);
    const cc = getCountryConfig(newCountry);
    // Zet standaard taal voor dat land
    if (cc.defaultLang.length > 0) setLang(cc.defaultLang[0]);
    setCountryState(newCountry);
  }, []);

  const formatCurrency = useCallback((amount) => {
    return new Intl.NumberFormat(config.locale, {
      style: 'currency',
      currency: config.currency,
    }).format(amount);
  }, [config]);

  return { t, lang, country, setLang, setCountry, config, formatCurrency };
}

// Auto-detect browsertaal
function detectBrowserLang() {
  const browserLang = navigator.language.split('-')[0];
  const supported = ['nl', 'fr', 'en', 'de'];
  return supported.includes(browserLang) ? browserLang : 'nl';
}

// buildpay/middleware/i18nMiddleware.js
// Express middleware — detecteert land + taal automatisch

const { COUNTRIES } = require('../i18n/countries');

// IP → land mapping via MaxMind GeoIP of Cloudflare CF-IPCountry header
function i18nMiddleware(req, res, next) {

  // 1. Land detecteren
  const countryFromHeader = req.headers['cf-ipcountry']; // Cloudflare
  const countryFromCookie = req.cookies?.['bp_country'];
  const country = countryFromCookie
    || (COUNTRIES[countryFromHeader] ? countryFromHeader : 'BE');

  // 2. Taal detecteren
  const langFromCookie = req.cookies?.['bp_lang'];
  const langFromHeader = req.headers['accept-language']
    ?.split(',')[0]?.split('-')[0];
  const supported = ['nl', 'fr', 'en', 'de'];
  const lang = langFromCookie
    || (supported.includes(langFromHeader) ? langFromHeader : 'nl');

  // 3. Attach aan request object
  req.i18n = { country, lang };

  // 4. SEO: stel juiste hreflang headers in
  res.setHeader('Content-Language', lang);
  res.setHeader('X-Buildpay-Country', country);

  next();
}

/**
 * URL structuur voor SEO:
 * buildpay.be/nl/klant         → NL, BE
 * buildpay.be/fr/client        → FR, BE
 * buildpay.nl/nl/aannemer      → NL, NL
 * buildpay.fr/fr/entrepreneur  → FR, FR
 * 
 * Subdomain strategie (later):
 * nl.buildpay.com → NL
 * fr.buildpay.com → FR
 * de.buildpay.com → DE
 */
function langRouter(req, res, next) {
  const [, langFromUrl] = req.path.split('/');
  if (['nl', 'fr', 'en', 'de'].includes(langFromUrl)) {
    req.i18n.lang = langFromUrl;
  }
  next();
}

// Hreflang tags genereren voor <head> (SEO)
function generateHreflangTags(baseUrl, currentPath) {
  const langs = [
    { lang: 'nl', country: 'BE', domain: 'buildpay.be' },
    { lang: 'fr', country: 'BE', domain: 'buildpay.be' },
    { lang: 'nl', country: 'NL', domain: 'buildpay.nl' },
    { lang: 'fr', country: 'FR', domain: 'buildpay.fr' },
    { lang: 'en', country: 'GB', domain: 'buildpay.co.uk' },
  ];

  return langs.map(({ lang, country, domain }) =>
    `<link rel="alternate" hreflang="${lang}-${country}" href="https://${domain}/${lang}${currentPath}">`
  ).join('\n');
}
