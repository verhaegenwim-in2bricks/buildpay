// buildpay/services/mangopay/setup.js
// Mangopay SDK initialisatie voor Buildpay

const Mangopay = require('mangopay2-nodejs-sdk');

// Initialiseer Mangopay client
const mangopay = new Mangopay({
  clientId:  process.env.MANGOPAY_CLIENT_ID,   // bv. 'buildpay_prod'
  clientApiKey: process.env.MANGOPAY_API_KEY,  // Geheime API sleutel
  baseUrl: process.env.NODE_ENV === 'production'
    ? 'https://api.mangopay.com'
    : 'https://api.sandbox.mangopay.com', // Sandbox voor development
});

// Buildpay commissie wallet ID (platform earnings)
const BUILDPAY_FEE_WALLET_ID = process.env.BUILDPAY_FEE_WALLET_ID;
const COMMISSION_RATE = 0.015; // 1.5%
const ARCHITECT_RATE = 0.0025; // 0.25%

module.exports = { mangopay, BUILDPAY_FEE_WALLET_ID, COMMISSION_RATE, ARCHITECT_RATE };

// buildpay/services/mangopay/createUsers.js
// Mangopay users en wallets aanmaken bij project start

const { mangopay } = require('./setup');

/**
 * Maak Mangopay Natural User aan voor particuliere klant
 */
async function createClientUser(client) {
  const user = await mangopay.Users.create({
    PersonType: 'NATURAL',
    FirstName:  client.firstName,
    LastName:   client.lastName,
    Email:      client.email,
    Birthday:  client.birthdayTimestamp,
    Nationality: 'BE',
    CountryOfResidence: 'BE',
  });

  // Maak escrow wallet aan voor klant
  const wallet = await mangopay.Wallets.create({
    Owners: [user.Id],
    Description: `Buildpay escrow — ${client.projectName}`,
    Currency: 'EUR',
  });

  return { userId: user.Id, walletId: wallet.Id };
}

/**
 * Maak Mangopay Legal User aan voor aannemer (bedrijf)
 */
async function createContractorUser(contractor) {
  const user = await mangopay.Users.create({
    PersonType: 'LEGAL',
    Name:       contractor.companyName,
    Email:      contractor.email,
    LegalPersonType: 'BUSINESS',
    LegalRepresentativeFirstName: contractor.repFirstName,
    LegalRepresentativeLastName:  contractor.repLastName,
    LegalRepresentativeEmail:     contractor.email,
    CompanyNumber: contractor.vatNumber, // BTW-nummer
    CountryOfResidence: 'BE',
  });

  const wallet = await mangopay.Wallets.create({
    Owners: [user.Id],
    Description: `Buildpay aannemer — ${contractor.companyName}`,
    Currency: 'EUR',
  });

  return { userId: user.Id, walletId: wallet.Id };
}

module.exports = { createClientUser, createContractorUser };

// buildpay/services/mangopay/escrowService.js
// Escrow storting via iDEAL / Bancontact / SEPA

const { mangopay } = require('./setup');

/**
 * Initieer een escrow PayIn via Bancontact (BE)
 * Klant wordt omgeleid naar bancontact betaalpagina
 */
async function createBancontactPayIn({ projectId, clientWalletId, amount, returnUrl }) {
  const payIn = await mangopay.PayIns.create({
    PaymentType: 'WEB',
    ExecutionType: 'WEB',
    PaymentDetails: { Style: 'REDIRECT' },
    CreditedWalletId: clientWalletId,   // → klant escrow wallet
    DebitedFunds: {
      Currency: 'EUR',
      Amount: amount * 100,   // Mangopay werkt in centen
    },
    Fees: { Currency: 'EUR', Amount: 0 }, // Geen fees bij storting
    ReturnURL: returnUrl,
    Tag: `buildpay_project_${projectId}`,
  });

  return {
    payInId:    payIn.Id,
    redirectUrl: payIn.RedirectURL, // Stuur klant hierheen
    status:     payIn.Status,
  };
}

/**
 * iDEAL PayIn voor Nederlandse klanten
 */
async function createIdealPayIn({ projectId, clientWalletId, amount, bank, returnUrl }) {
  const payIn = await mangopay.PayIns.create({
    PaymentType: 'IDEAL',
    ExecutionType: 'WEB',
    Bic: bank,                 // bv. 'INGBNL2A' voor ING
    CreditedWalletId: clientWalletId,
    DebitedFunds: {
      Currency: 'EUR',
      Amount: amount * 100,
    },
    Fees: { Currency: 'EUR', Amount: 0 },
    ReturnURL: returnUrl,
    Tag: `buildpay_project_${projectId}`,
  });

  return { payInId: payIn.Id, redirectUrl: payIn.RedirectURL };
}

module.exports = { createBancontactPayIn, createIdealPayIn };

// buildpay/services/mangopay/releaseEscrow.js
// Milestone vrijgeven: Transfer + PayOut naar bankrekening

const { mangopay, BUILDPAY_FEE_WALLET_ID, COMMISSION_RATE, ARCHITECT_RATE } = require('./setup');

/**
 * Geef milestone vrij na goedkeuring door klant én architect
 * 
 * Flow:
 * 1. Transfer van klant wallet → aannemer wallet (min commissie)
 * 2. Transfer commissie → Buildpay fee wallet
 * 3. PayOut aannemer wallet → bankrekening (SEPA)
 * 4. Optioneel: architect vergoeding
 */
async function releaseMilestone({
  milestoneId, milestoneAmount,
  clientWalletId, contractorWalletId,
  contractorBankAccountId,
  architectWalletId,   // null als geen architect
  approvedByClient, approvedByArchitect
}) {

  // Validatie: beide partijen moeten goedkeuren
  if (!approvedByClient) throw new Error('Klant heeft nog niet goedgekeurd');
  if (architectWalletId && !approvedByArchitect) {
    throw new Error('Architect heeft nog niet gevalideerd');
  }

  const amountInCents = milestoneAmount * 100;

  // Bereken commissies
  const buildpayFee = Math.round(amountInCents * COMMISSION_RATE);
  const architectFee = architectWalletId
    ? Math.round(amountInCents * ARCHITECT_RATE)
    : 0;
  const netAmount = amountInCents - buildpayFee - architectFee;

  // Stap 1: Transfer klant → aannemer (netto bedrag)
  const transfer = await mangopay.Transfers.create({
    AuthorId:       clientWalletId,    // klant is eigenaar
    DebitedWalletId: clientWalletId,
    CreditedWalletId: contractorWalletId,
    DebitedFunds:  { Currency: 'EUR', Amount: netAmount },
    Fees:          { Currency: 'EUR', Amount: buildpayFee },
    Tag: `milestone_${milestoneId}_release`,
  });

  // Stap 2 (optioneel): Architect vergoeding
  if (architectWalletId && architectFee > 0) {
    await mangopay.Transfers.create({
      AuthorId:        clientWalletId,
      DebitedWalletId:  clientWalletId,
      CreditedWalletId: architectWalletId,
      DebitedFunds: { Currency: 'EUR', Amount: architectFee },
      Fees:         { Currency: 'EUR', Amount: 0 },
      Tag: `milestone_${milestoneId}_architect_fee`,
    });
  }

  // Stap 3: PayOut aannemer → bankrekening (SEPA)
  const payout = await mangopay.PayOuts.create({
    AuthorId:        contractorWalletId,
    DebitedWalletId:  contractorWalletId,
    DebitedFunds:    { Currency: 'EUR', Amount: netAmount },
    Fees:            { Currency: 'EUR', Amount: 0 },
    BankAccountId:   contractorBankAccountId,  // IBAN aannemer
    BankWireRef:     `Buildpay MS-${milestoneId}`,
    Tag: `milestone_${milestoneId}_payout`,
  });

  return {
    transferId: transfer.Id,
    payoutId:   payout.Id,
    netAmount:  netAmount / 100,
    buildpayFee: buildpayFee / 100,
    architectFee: architectFee / 100,
    estimatedArrival: 'binnen 24 uur',
  };
}

/**
 * Terugbetaling bij geschil — refund naar klant
 */
async function refundToClient({ payInId, amount, reason }) {
  const refund = await mangopay.PayIns.createRefund(payInId, {
    AuthorId: payInId,
    DebitedFunds: { Currency: 'EUR', Amount: amount * 100 },
    Fees: { Currency: 'EUR', Amount: 0 },
    Tag: reason,
  });
  return { refundId: refund.Id, status: refund.Status };
}

module.exports = { releaseMilestone, refundToClient };

// buildpay/routes/webhooks.js
// Mangopay webhook handler — verwerkt betalingsevents

const express = require('express');
const router  = express.Router();
const { mangopay } = require('../services/mangopay/setup');
const ProjectService = require('../services/projectService');
const NotificationService = require('../services/notificationService');

// POST /webhooks/mangopay
router.post('/mangopay', async (req, res) => {
  const { EventType, ResourceId, Date } = req.body;

  log(`Webhook: ${EventType} — ${ResourceId}`);

  switch (EventType) {

    case 'PAYIN_NORMAL_SUCCEEDED': {
      // Escrow storting gelukt → project kan starten
      const payIn = await mangopay.PayIns.get(ResourceId);
      const projectId = payIn.Tag.replace('buildpay_project_', '');

      await ProjectService.markEscrowReceived(projectId, {
        amount: payIn.CreditedFunds.Amount / 100,
        payInId: payIn.Id,
      });

      await NotificationService.send({
        type: 'ESCROW_RECEIVED',
        projectId,
        message: `Escrow gestort: €${payIn.CreditedFunds.Amount / 100}`,
      });
      break;
    }

    case 'PAYIN_NORMAL_FAILED': {
      // Storting mislukt → klant notificeren
      const payIn = await mangopay.PayIns.get(ResourceId);
      await NotificationService.send({
        type: 'PAYIN_FAILED',
        projectId: payIn.Tag.replace('buildpay_project_', ''),
        message: `Betaling mislukt: ${payIn.ResultMessage}`,
      });
      break;
    }

    case 'TRANSFER_NORMAL_SUCCEEDED': {
      // Milestone transfer gelukt
      const transfer = await mangopay.Transfers.get(ResourceId);
      const [, milestoneId] = transfer.Tag.split('_');
      await ProjectService.markMilestoneTransferred(milestoneId);
      break;
    }

    case 'PAYOUT_NORMAL_SUCCEEDED': {
      // Uitbetaling naar aannemer gelukt
      const payout = await mangopay.PayOuts.get(ResourceId);
      const [, msId] = payout.Tag.split('_');
      await ProjectService.markMilestonePaid(msId);
      await NotificationService.send({
        type: 'MILESTONE_PAID',
        milestoneId: msId,
        message: `€${payout.DebitedFunds.Amount / 100} uitbetaald aan aannemer`,
      });
      break;
    }

    case 'KYC_SUCCEEDED': {
      // KYC verificatie geslaagd
      await ProjectService.markKycVerified(ResourceId);
      break;
    }
  }

  res.status(200).json({ received: true });
});

module.exports = router;

// buildpay/services/mangopay/kycService.js
// KYC verificatie voor aannemers

const { mangopay } = require('./setup');
const fs = require('fs');

/**
 * Upload KYC document (identiteitskaart, KBO-uittreksel, etc.)
 * Mangopay verwerkt dit automatisch via hun verificatiesysteem
 */
async function uploadKycDocument({ userId, documentType, filePath }) {
  // Stap 1: Maak KYC document object aan
  const kycDoc = await mangopay.Users.createKycDocument(userId, {
    Type: documentType,
    // IDENTITY_PROOF | REGISTRATION_PROOF | ARTICLES_OF_ASSOCIATION
  });

  // Stap 2: Upload pagina's
  const fileContent = fs.readFileSync(filePath).toString('base64');
  await mangopay.Users.createKycPage(userId, kycDoc.Id, {
    File: fileContent,
  });

  // Stap 3: Verstuur ter verificatie
  const submitted = await mangopay.Users.updateKycDocument(userId, {
    Id: kycDoc.Id,
    Status: 'VALIDATION_ASKED',
  });

  return {
    kycDocId: submitted.Id,
    status:   submitted.Status, // VALIDATION_ASKED | VALIDATED | REFUSED
  };
}

/**
 * Vereiste KYC documenten per gebruikerstype
 */
const KYC_REQUIREMENTS = {
  NATURAL: [                       // Particuliere klant
    'IDENTITY_PROOF',              // Identiteitskaart
  ],
  LEGAL_BUSINESS: [               // Aannemer (BV/BVBA)
    'REGISTRATION_PROOF',          // KBO-uittreksel
    'ARTICLES_OF_ASSOCIATION',     // Statuten
    'IDENTITY_PROOF',              // ID zaakvoerder
  ],
};

/**
 * Controleer KYC status van een gebruiker
 */
async function checkKycStatus(userId) {
  const docs = await mangopay.Users.getKycDocuments(userId);
  const validated = docs.filter(d => d.Status === 'VALIDATED');
  return {
    isFullyVerified: validated.length > 0,
    documents: docs.map(d => ({
      type:   d.Type,
      status: d.Status,
    })),
  };
}

module.exports = { uploadKycDocument, checkKycStatus, KYC_REQUIREMENTS };
